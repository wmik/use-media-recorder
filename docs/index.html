import useMediaRecorder from '@wmik/use-media-recorder';

function RecorderApp() {
  const {
    status,
    mediaBlob,
    startRecording,
    stopRecording
  } = useMediaRecorder({
    recordScreen: true,
    blobOptions: { type: 'video/webm' }
  });

  return (
    <div>
      <button onClick={startRecording}>
        Start Recording
      </button>
    </div>
  );
}

Code Example

const {
  error,
  status,
  mediaBlob,
  stopRecording,
  getMediaStream,
  startRecording
} = useMediaRecorder({
  recordScreen: true,
  blobOptions: { 
    type: 'video/webm' 
  },
  mediaStreamConstraints: { 
    audio: true, 
    video: true 
  }
});

// Available status values:
// 'idle', 'acquiring_media', 'ready',
// 'recording', 'paused', 'stopping',
// 'stopped', 'failed'

const handleStart = async () => {
  await getMediaStream();
  startRecording();
};

const handleStop = () => {
  stopRecording();
};

// Use mediaBlob when recording stops
useEffect(() => {
  if (mediaBlob) {
    const url = URL.createObjectURL(mediaBlob);
    // Use the recorded media...
  }
}, [mediaBlob]);
